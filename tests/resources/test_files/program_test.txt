# تعريف الثوابت
عدد_صحيح ثابت حجم_المصفوفة = 5.

# تعريف المصفوفة
عدد_صحيح مصفوفة[5].

# دالة لحساب مجموع المصفوفة
دالة مجموع_المصفوفة(عدد_صحيح* مصف, عدد_صحيح طول) {
    عدد_صحيح مجموع = 0.
    عدد_صحيح مؤشر = 0.
    
    طالما (مؤشر < طول) {
        مجموع = مجموع + مصف[مؤشر].
        مؤشر = مؤشر + 1.
    }
    
    إرجع مجموع.
}

# دالة لإيجاد أكبر قيمة في المصفوفة
دالة أكبر_قيمة(عدد_صحيح* مصف, عدد_صحيح طول) {
    إذا (طول == 0) {
        إرجع 0.
    }
    
    عدد_صحيح أكبر = مصف[0].
    عدد_صحيح مؤشر = 1.
    
    طالما (مؤشر < طول) {
        إذا (مصف[مؤشر] > أكبر) {
            أكبر = مصف[مؤشر].
        }
        مؤشر = مؤشر + 1.
    }
    
    إرجع أكبر.
}

# الدالة الرئيسية
دالة رئيسية() {
    # تهيئة المصفوفة
    مصفوفة[0] = 10.
    مصفوفة[1] = 20.
    مصفوفة[2] = 30.
    مصفوفة[3] = 40.
    مصفوفة[4] = 50.
    
    # حساب وطباعة المجموع
    عدد_صحيح مجموع = مجموع_المصفوفة(مصفوفة, حجم_المصفوفة).
    اطبع("مجموع المصفوفة: %d\n", مجموع).
    
    # إيجاد وطباعة أكبر قيمة
    عدد_صحيح أكبر = أكبر_قيمة(مصفوفة, حجم_المصفوفة).
    اطبع("أكبر قيمة في المصفوفة: %d\n", أكبر).
    
    إرجع 0.
}
